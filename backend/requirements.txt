# backend/requirements.txt
#
# Pin to the minor version (X.Y.*) for reproducibility across environments.
# Upgrade deliberately via: pip install --upgrade <package> and re-pin.
#
# ─── Core ORM & Database Driver ──────────────────────────────────────────────

# SQLAlchemy 2.x — modern async-ready ORM and Core.
# The 2.x series introduces typed mapped_column(), improved session management,
# and first-class async support (asyncpg driver).
sqlalchemy>=2.0,<3.0

# GeoAlchemy2 — SQLAlchemy extension for PostGIS spatial types.
# Provides the Geometry column type, WKB/WKT element handling, and
# automatic spatial index creation.
geoalchemy2>=0.14,<1.0

# psycopg2-binary — PostgreSQL adapter for Python (pre-compiled wheels).
# "-binary" bundles libpq, avoiding the need for system-level pg_config /
# PostgreSQL development headers. Switch to `psycopg2` (compiled from source)
# in production Docker images for better performance and smaller attack surface.
psycopg2-binary>=2.9,<3.0

# ─── Environment & Configuration ─────────────────────────────────────────────

# python-dotenv — Loads key=value pairs from a .env file into os.environ.
# Keeps secrets out of source code and makes the app portable across
# local, CI, staging, and production environments.
python-dotenv>=1.0,<2.0

# ─── Web Framework & API ──────────────────────────────────────────────────────

# FastAPI — high-performance async web framework built on Starlette + Pydantic.
# Chosen for: automatic OpenAPI spec generation, dependency injection, native
# async support, and Pydantic v2 integration.  The ingestion endpoint uses
# async def handlers to avoid blocking the event loop during DB writes.
fastapi>=0.111,<1.0

# uvicorn[standard] — ASGI server that runs the FastAPI application.
# The [standard] extras install uvloop (faster event loop) and httptools
# (faster HTTP parsing), giving ~20% throughput improvement over the base build.
# In production, run behind Gunicorn: gunicorn -k uvicorn.workers.UvicornWorker
uvicorn[standard]>=0.29,<1.0

# Pydantic v2 — data validation and serialisation library.
# The v2 core is rewritten in Rust, providing 5–50x faster validation than v1.
# Used for TelemetryReadingIn / IngestionResponse models in the ingestion API.
# Key v2 patterns: model_validate(), model_dump(), ConfigDict(), Field(ge=, le=).
pydantic>=2.7,<3.0

# httpx — async HTTP client for the sensor simulator and stress tests.
# Unlike `requests` (sync only), httpx supports asyncio via AsyncClient.
# Provides connection pooling (Limits) and configurable timeouts out of the box.
httpx>=0.27,<1.0

# anyio — async compatibility layer; required by httpx and FastAPI test clients.
anyio>=4.0,<5.0

# numpy — Required for numerical operations in predictive risk heuristics
# and later ML integration.
numpy>=1.26.0

# ─── Future placeholders (uncomment when needed) ──────────────────────────────
# alembic>=1.13,<2.0     # Schema migration tool (ALTER TABLE with rollback support)

# asyncpg — Full async PostgreSQL driver for the CRUD API layer.
# asyncpg implements the PostgreSQL binary wire protocol natively, without
# using libpq (the C client library that psycopg2 depends on).  This gives:
#   • True async I/O — no thread pool wrapper required
#   • 3–5x faster query execution vs psycopg2 for read-heavy workloads
#   • Native PostgreSQL type support (JSONB, UUID, arrays, PostGIS binary)
# Used by: database/session.py → async_sessionmaker → CRUD routes
asyncpg>=0.29,<1.0

# pydantic-settings — Settings management for Pydantic v2 apps.
# Extends BaseModel with environment variable and .env file reading.
# Provides type coercion, validation, and nested config support.
# Used by: core/config.py → Settings class → get_settings() singleton
pydantic-settings>=2.0,<3.0

# greenlet — Required by SQLAlchemy async support.
# SQLAlchemy's async engine uses greenlet internally to bridge the
# synchronous ORM internals with the asyncio event loop.  This is an
# implementation detail — applications don't call greenlet directly.
greenlet>=3.0,<4.0

# ─── DEM / Geospatial Raster Processing ──────────────────────────────────────

# rasterio — high-level Pythonic interface to GDAL for reading/writing raster
# formats (GeoTIFF, NetCDF, HDF5, etc.).  Chosen over raw GDAL bindings because:
#   • Context-manager API guarantees file handle cleanup on exceptions.
#   • Native NumPy integration for array-level raster operations.
#   • Windowed reads allow single-pixel access without loading entire rasters.
#   • Pre-built wheels on PyPI for all platforms (no system GDAL install needed).
rasterio>=1.3,<2.0

# numpy — fundamental array library for rasterio pixel data.
# rasterio.read() returns NumPy arrays; elevation values are extracted via
# standard array indexing (no loops needed for bulk operations).
numpy>=1.24,<3.0

# pyproj — Python bindings for the PROJ coordinate transformation library.
# Used to transform sensor WGS84 (EPSG:4326) coordinates into the DEM's
# native CRS before raster row/column lookup.  Essential for DEMs stored in
# projected CRS (e.g., UTM Zone 43N for Indian subcontinent datasets).
pyproj>=3.6,<4.0

# ASGITransport(app=app) sends requests directly to the ASGI app in-process
# without a network socket, enabling fast, realistic integration tests.
